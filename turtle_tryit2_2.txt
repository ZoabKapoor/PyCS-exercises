import turtle

wn = turtle.Screen()
wn.bgcolor("black")        # set the window background color

sarah = turtle.Turtle()
sarah.color("green")
sarah.pensize(3)

for side in range(3):
    sarah.forward(45)
    sarah.pensize(7)
    sarah.forward(45)
    sarah.pensize(3)
    sarah.forward(45)
    sarah.pensize(7)
    sarah.forward(45)
    sarah.pensize(3)
    
    sarah.left(120)

wn.exitonclick()                # wait for a user click on the canvas
